# CPA Usage Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add opt-in CLIProxyAPI request-level usage visibility without changing the existing `Provider usage` quota panel.

**Architecture:** Keep CPA request usage as its own backend/client/storage/UI path. Non-destructive `/api-key-usage` powers passive status; destructive `/usage-queue` is consumed only by an explicit manual endpoint or opt-in background collector and persisted into deck SQLite.

**Tech Stack:** Bun, Hono, Bun SQLite, React, TypeScript, `@omp-deck/protocol`, existing Settings env schema and StatusPanel UI patterns.

---

```ts
export interface CpaManagementClientOptions {
	baseUrl: string;
	managementKey: string;
	fetcher?: typeof fetch;
}

export interface ApiKeyUsageBucket {
	provider: string;
	success: number;
	failed: number;
}

function isRecord(value: unknown): value is Record<string, unknown> {
	return typeof value === "object" && value !== null && !Array.isArray(value);
}

function numberOrZero(value: unknown): number {
	return typeof value === "number" && Number.isFinite(value) ? value : 0;
}

function stringOrEmpty(value: unknown): string {
	return typeof value === "string" ? value : "";
}

export function normalizeApiKeyUsage(value: unknown): ApiKeyUsageBucket[] {
	const rows = isRecord(value) && Array.isArray(value.providers) ? value.providers : [];
	return rows.filter(isRecord).map((row) => ({
		provider: stringOrEmpty(row.provider),
		success: numberOrZero(row.success ?? row.success_count ?? row.successCount),
		failed: numberOrZero(row.failed ?? row.failed_count ?? row.failedCount),
	})).filter((row) => row.provider.length > 0);
}

export function normalizeUsageQueue(value: unknown): Record<string, unknown>[] {
	const rows = Array.isArray(value)
		? value
		: isRecord(value) && Array.isArray(value.records)
			? value.records
			: isRecord(value) && Array.isArray(value.items)
				? value.items
				: [];
	return rows.filter(isRecord);
}

export class CpaManagementClient {
	readonly #baseUrl: string;
	readonly #managementKey: string;
	readonly #fetcher: typeof fetch;

	constructor(opts: CpaManagementClientOptions) {
		this.#baseUrl = opts.baseUrl.replace(/\/+$/, "");
		this.#managementKey = opts.managementKey;
		this.#fetcher = opts.fetcher ?? fetch;
	}

	async getApiKeyUsage(): Promise<unknown> {
		return await this.#getJson("/v0/management/api-key-usage");
	}

	async popUsageQueue(count: number): Promise<unknown> {
		return await this.#getJson(`/v0/management/usage-queue?count=${encodeURIComponent(String(count))}`);
	}

	async #getJson(path: string): Promise<unknown> {
		const headers = new Headers();
		headers.set("authorization", `Bearer ${this.#managementKey}`);
		const res = await this.#fetcher(`${this.#baseUrl}${path}`, { headers });
		if (!res.ok) throw new Error(`CPA Management API ${res.status}: ${await res.text()}`);
		return await res.json();
	}
}
```

```ts
import { createHash } from "node:crypto";
import type { CpaUsageEventsResponse, CpaUsageEventWire, CpaUsageSummaryResponse, CpaUsageSummaryRow } from "@omp-deck/protocol";
import { getDb, nowIso } from "./db/index.ts";

export type CpaUsageWindow = "1h" | "24h" | "7d" | "30d";

const WINDOW_MS: Record<CpaUsageWindow, number> = {
	"1h": 60 * 60 * 1000,
	"24h": 24 * 60 * 60 * 1000,
	"7d": 7 * 24 * 60 * 60 * 1000,
	"30d": 30 * 24 * 60 * 60 * 1000,
};

function isRecord(value: unknown): value is Record<string, unknown> {
	return typeof value === "object" && value !== null && !Array.isArray(value);
}

function optionalString(value: unknown): string | null {
	return typeof value === "string" && value.length > 0 ? value : null;
}

function optionalNumber(value: unknown): number | null {
	return typeof value === "number" && Number.isFinite(value) ? value : null;
}

function boolInt(value: unknown): number {
	return value === true || value === 1 ? 1 : 0;
}

function stableId(record: Record<string, unknown>, receivedAt: string): string {
	const id = optionalString(record.id ?? record.request_id ?? record.requestId);
	if (id) return id;
	return createHash("sha256").update(JSON.stringify(record)).update(receivedAt.slice(0, 16)).digest("hex").slice(0, 32);
}

function wireFromRow(row: Record<string, unknown>): CpaUsageEventWire {
	return {
		id: String(row.id),
		receivedAt: String(row.received_at),
		requestedAt: optionalString(row.requested_at) ?? undefined,
		provider: optionalString(row.provider) ?? undefined,
		model: optionalString(row.model) ?? undefined,
		alias: optionalString(row.alias) ?? undefined,
		apiKeyHash: optionalString(row.api_key_hash) ?? undefined,
		authId: optionalString(row.auth_id) ?? undefined,
		authIndex: optionalString(row.auth_index) ?? undefined,
		authType: optionalString(row.auth_type) ?? undefined,
		source: optionalString(row.source) ?? undefined,
		latencyMs: optionalNumber(row.latency_ms) ?? undefined,
		ttftMs: optionalNumber(row.ttft_ms) ?? undefined,
		failed: row.failed === 1,
		statusCode: optionalNumber(row.status_code) ?? undefined,
		inputTokens: optionalNumber(row.input_tokens) ?? undefined,
		outputTokens: optionalNumber(row.output_tokens) ?? undefined,
		totalTokens: optionalNumber(row.total_tokens) ?? undefined,
	};
}

export function insertCpaUsageRecords(records: Record<string, unknown>[]): { seen: number; inserted: number } {
	const db = getDb();
	const receivedAt = nowIso();
	const stmt = db.prepare(`INSERT OR IGNORE INTO cpa_usage_events (
		id, received_at, requested_at, provider, model, alias, api_key_hash, auth_id, auth_index, auth_type, source,
		latency_ms, ttft_ms, failed, status_code, input_tokens, output_tokens, reasoning_tokens, cached_tokens,
		cache_read_tokens, cache_creation_tokens, total_tokens, raw_json
	) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`);
	let inserted = 0;
	db.transaction(() => {
		for (const record of records) {
			if (!isRecord(record)) continue;
			const result = stmt.run(
				stableId(record, receivedAt),
				receivedAt,
				optionalString(record.requested_at ?? record.requestedAt ?? record.timestamp),
				optionalString(record.provider),
				optionalString(record.model),
				optionalString(record.alias),
				optionalString(record.api_key_hash ?? record.apiKeyHash),
				optionalString(record.auth_id ?? record.authId),
				optionalString(record.auth_index ?? record.authIndex),
				optionalString(record.auth_type ?? record.authType),
				optionalString(record.source),
				optionalNumber(record.latency_ms ?? record.latencyMs),
				optionalNumber(record.ttft_ms ?? record.ttftMs),
				boolInt(record.failed),
				optionalNumber(record.status_code ?? record.statusCode),
				optionalNumber(record.input_tokens ?? record.inputTokens),
				optionalNumber(record.output_tokens ?? record.outputTokens),
				optionalNumber(record.reasoning_tokens ?? record.reasoningTokens),
				optionalNumber(record.cached_tokens ?? record.cachedTokens),
				optionalNumber(record.cache_read_tokens ?? record.cacheReadTokens),
				optionalNumber(record.cache_creation_tokens ?? record.cacheCreationTokens),
				optionalNumber(record.total_tokens ?? record.totalTokens),
				JSON.stringify(record),
			);
			if (result.changes > 0) inserted += 1;
		}
	})();
	return { seen: records.length, inserted };
}

export function listCpaUsageEvents(opts: { limit: number; provider?: string; model?: string }): CpaUsageEventsResponse {
	const limit = Math.max(1, Math.min(500, opts.limit));
	const clauses: string[] = [];
	const params: unknown[] = [];
	if (opts.provider) { clauses.push("provider = ?"); params.push(opts.provider); }
	if (opts.model) { clauses.push("model = ?"); params.push(opts.model); }
	const where = clauses.length > 0 ? `WHERE ${clauses.join(" AND ")}` : "";
	const rows = getDb().query<Record<string, unknown>, unknown[]>(`SELECT * FROM cpa_usage_events ${where} ORDER BY received_at DESC LIMIT ?`).all(...params, limit);
	return { events: rows.map(wireFromRow) };
}

export function summarizeCpaUsage(window: CpaUsageWindow): CpaUsageSummaryResponse {
	const since = new Date(Date.now() - WINDOW_MS[window]).toISOString();
	const rows = getDb().query<Record<string, unknown>, [string]>(`
		SELECT provider, model, COUNT(*) AS requests, SUM(failed) AS failures,
		SUM(COALESCE(input_tokens, 0)) AS inputTokens,
		SUM(COALESCE(output_tokens, 0)) AS outputTokens,
		SUM(COALESCE(total_tokens, 0)) AS totalTokens,
		AVG(latency_ms) AS avgLatencyMs,
		AVG(ttft_ms) AS avgTtftMs
		FROM cpa_usage_events
		WHERE received_at >= ?
		GROUP BY provider, model
		ORDER BY requests DESC
	`).all(since);
	return {
		window,
		generatedAt: nowIso(),
		status: { configured: false, collectorEnabled: false, mode: "off", managementReachable: false },
		rows: rows.map((row): CpaUsageSummaryRow => ({
			provider: String(row.provider ?? "unknown"),
			model: String(row.model ?? "unknown"),
			requests: Number(row.requests ?? 0),
			failures: Number(row.failures ?? 0),
			inputTokens: Number(row.inputTokens ?? 0),
			outputTokens: Number(row.outputTokens ?? 0),
			totalTokens: Number(row.totalTokens ?? 0),
			avgLatencyMs: optionalNumber(row.avgLatencyMs) ?? undefined,
			avgTtftMs: optionalNumber(row.avgTtftMs) ?? undefined,
		})),
	};
}
```

```ts
import { Hono } from "hono";
import type { CpaUsageRunOnceResponse, CpaUsageStatusResponse } from "@omp-deck/protocol";
import type { Config } from "./config.ts";
import { CpaManagementClient, normalizeUsageQueue } from "./cpa-management-client.ts";
import { insertCpaUsageRecords, listCpaUsageEvents, summarizeCpaUsage, type CpaUsageWindow } from "./cpa-usage-store.ts";

const QUEUE_WARNING = "This consumes CLIProxyAPI usage-queue records. Do not enable if another CPA usage collector is already running.";

function configured(config: Config): boolean {
	return Boolean(config.cpaManagementUrl && config.cpaManagementKey);
}

export function buildCpaUsageStatus(config: Config, last?: { pollAt?: string; error?: string; reachable?: boolean }): CpaUsageStatusResponse {
	const isConfigured = configured(config);
	return {
		configured: isConfigured,
		collectorEnabled: config.cpaUsageCollectorEnabled,
		mode: config.cpaUsageCollectorEnabled ? "usage-queue" : isConfigured ? "api-key-usage" : "off",
		managementReachable: last?.reachable ?? false,
		lastPollAt: last?.pollAt,
		lastError: last?.error,
		warning: config.cpaUsageCollectorEnabled ? QUEUE_WARNING : undefined,
	};
}

function parseWindow(raw: string | undefined): CpaUsageWindow {
	return raw === "1h" || raw === "7d" || raw === "30d" ? raw : "24h";
}

function clientFromConfig(config: Config): CpaManagementClient {
	if (!config.cpaManagementUrl || !config.cpaManagementKey) throw new Error("CPA Management API is not configured");
	return new CpaManagementClient({ baseUrl: config.cpaManagementUrl, managementKey: config.cpaManagementKey });
}

export function buildCpaUsageRouter(config: Config): Hono {
	const app = new Hono();

	app.get("/cpa-usage/status", (c) => c.json(buildCpaUsageStatus(config)));

	app.get("/cpa-usage/summary", (c) => {
		const body = summarizeCpaUsage(parseWindow(c.req.query("window")));
		return c.json({ ...body, status: buildCpaUsageStatus(config) });
	});

	app.get("/cpa-usage/events", (c) => {
		const limit = Number.parseInt(c.req.query("limit") ?? "100", 10);
		return c.json(listCpaUsageEvents({
			limit: Number.isFinite(limit) ? limit : 100,
			provider: c.req.query("provider") || undefined,
			model: c.req.query("model") || undefined,
		}));
	});

	app.post("/cpa-usage/collector/run-once", async (c) => {
		try {
			const client = clientFromConfig(config);
			const raw = await client.popUsageQueue(100);
			const records = normalizeUsageQueue(raw);
			const result = insertCpaUsageRecords(records);
			const body: CpaUsageRunOnceResponse = {
				...result,
				status: buildCpaUsageStatus(config, { pollAt: new Date().toISOString(), reachable: true }),
			};
			return c.json(body);
		} catch (err) {
			if (String(err).includes("not configured")) return c.json({ error: "CPA Management API is not configured" }, 400);
			return c.json({ error: String(err) }, 502);
		}
	});

	return app;
}
```

```ts
import type { Timer } from "bun";
import type { Config } from "./config.ts";
import { CpaManagementClient, normalizeUsageQueue } from "./cpa-management-client.ts";
import { insertCpaUsageRecords } from "./cpa-usage-store.ts";
import { logger } from "./log.ts";

const log = logger("cpa-usage-runner");
const POLL_MS = 60_000;

export function shouldRunCpaCollector(config: Config): boolean {
	return Boolean(config.cpaUsageCollectorEnabled && config.cpaManagementUrl && config.cpaManagementKey);
}

export class CpaUsageRunner {
	readonly #config: Config;
	#timer: Timer | undefined;
	#stopped = true;
	#backoffMs = POLL_MS;

	constructor(config: Config) {
		this.#config = config;
	}

	start(): void {
		if (!shouldRunCpaCollector(this.#config)) return;
		this.#stopped = false;
		this.#schedule(0);
	}

	stop(): void {
		this.#stopped = true;
		if (this.#timer) clearTimeout(this.#timer);
		this.#timer = undefined;
	}

	#schedule(delay: number): void {
		this.#timer = setTimeout(() => void this.#poll(), delay);
	}

	async #poll(): Promise<void> {
		if (this.#stopped || !this.#config.cpaManagementUrl || !this.#config.cpaManagementKey) return;
		try {
			const client = new CpaManagementClient({ baseUrl: this.#config.cpaManagementUrl, managementKey: this.#config.cpaManagementKey });
			const raw = await client.popUsageQueue(100);
			const result = insertCpaUsageRecords(normalizeUsageQueue(raw));
			if (result.inserted > 0) log.info(`stored ${result.inserted} CPA usage event(s)`);
			this.#backoffMs = POLL_MS;
		} catch (err) {
			log.warn("CPA usage collector poll failed", err);
			this.#backoffMs = Math.min(this.#backoffMs * 2, 10 * POLL_MS);
		}
		if (!this.#stopped) this.#schedule(this.#backoffMs);
	}
}
```
